            <div class="flex-grow relative canvas-bg rounded-xl border-2 border-stone-300 overflow-hidden" id="canvas">

                <div v-for="el in elements" 

                     :key="el.id"

                     :id="'el-' + el.id"

                     class="element-item group"

                     :style="{ width: el.w + 'px', height: el.h + 'px', left: el.x + 'px', top: el.y + 'px', zIndex: el.z, opacity: el.opacity }">

                    

                    <img :src="el.img" :alt="el.name" class="pointer-events-none">

                    

                    <div class="resize-handle absolute -bottom-1 -right-1 w-6 h-6 bg-stone-800/30 rounded-full opacity-0 transition-opacity cursor-nwse-resize border-2 border-white"></div>

                </div>

            <aside class="w-72 bg-white/80 backdrop-blur-md p-6 rounded-xl shadow-xl border border-stone-200 overflow-y-auto">

                <h2 class="font-bold text-stone-800 mb-6 border-b pb-2 tracking-widest text-center">物件圖層</h2>

                <div class="space-y-4">

                    <div v-for="el in sortedElements" :key="el.id" class="p-3 bg-white rounded-lg shadow-sm border border-stone-100 flex flex-col gap-2">

                        <div class="flex items-center justify-between">

                            <span class="text-sm font-bold text-stone-700">{{ el.name }}</span>

                            <div class="flex gap-1">

                                <button @click="adjustZ(el.id, 1)" class="w-6 h-6 bg-stone-100 rounded hover:bg-stone-200 text-xs">▲</button>

                                <button @click="adjustZ(el.id, -1)" class="w-6 h-6 bg-stone-100 rounded hover:bg-stone-200 text-xs">▼</button>

                            </div>

                        </div>

                            <input type="range" min="0.1" max="1" step="0.1" v-model="el.opacity" class="flex-grow h-1 bg-stone-200 rounded-lg appearance-none cursor-pointer">

                        </div>

    <script>

        const { createApp } = Vue;

        createApp({

            data() {

                return {

                    elements: [

                        // 注意：這裡的 img 路徑要與你 GitHub 上的圖片檔名完全一致

                        { id: 1, name: '遠山', x: 50, y: 50, w: 400, h: 250, z: 1, opacity: 0.6, img: 'mountain.png' },

                        { id: 2, name: '近石', x: 100, y: 450, w: 150, h: 100, z: 10, opacity: 1, img: 'rock.png' },

                        { id: 3, name: '古樹', x: 500, y: 300, w: 100, h: 200, z: 8, opacity: 1, img: 'tree.png' },

                        { id: 4, name: '小船', x: 250, y: 400, w: 100, h: 40, z: 5, opacity: 0.8, img: 'boat.png' },

                        { id: 5, name: '小橋', x: 400, y: 500, w: 180, h: 60, z: 6, opacity: 1, img: 'bridge.png' }

                    ]

                }

            },

            computed: { sortedElements() { return [...this.elements].sort((a, b) => b.z - a.z); } },

            mounted() { this.initInteract(); },

            methods: {

                initInteract() {

                    interact('.element-item').draggable({

                        listeners: { move: (event) => {

                            const id = event.target.id.replace('el-', '');

                            const el = this.elements.find(e => e.id == id);

                            el.x += event.dx; el.y += event.dy;

                        }}

                    }).resizable({

                        edges: { right: true, bottom: true, bottomRight: true },

                        listeners: { move: (event) => {

                            const id = event.target.id.replace('el-', '');

                            const el = this.elements.find(e => e.id == id);

                            el.w = event.rect.width; el.h = event.rect.height;

                            el.x += event.deltaRect.left; el.y += event.deltaRect.top;

                        }}

                    });

                },

                adjustZ(id, delta) { const el = this.elements.find(e => e.id === id); el.z += delta; },

                reset() { location.reload(); }

            }

        }).mount('#app');

    </script>
